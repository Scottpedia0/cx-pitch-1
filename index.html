    <script>

        document.addEventListener('DOMContentLoaded', function () {

            // --- Elements ---

            const slides = document.querySelectorAll('.slide');

            const prevBtn = document.getElementById('prevBtn');

            const nextBtn = document.getElementById('nextBtn');

            const slideCounter = document.getElementById('slideCounter');

            const totalSlides = slides.length;

            let currentSlide = 1;


            // --- Functions ---

            function updateSlides() {

                slides.forEach(slide => {

                    const slideNumber = parseInt(slide.dataset.slide, 10);

                    slide.classList.remove('active');

                    if (slideNumber === currentSlide) {

                        slide.classList.add('active');

                        // Trigger animations for elements on the active slide

                        triggerAnimations(slide);

                    }

                });

                

                // Update counter and button states

                slideCounter.textContent = `${currentSlide} / ${totalSlides}`;

                prevBtn.disabled = currentSlide === 1;

                nextBtn.disabled = currentSlide === totalSlides;

            }


            function triggerAnimations(slide) {

                const animatables = slide.querySelectorAll('.animatable');

                // Reset animations first

                animatables.forEach(el => {

                    el.classList.remove('fade-in', 'slide-in-from-bottom', 'zoom-in', 'animate');

                    void el.offsetWidth; // Force reflow to restart animation

                });


                // Apply animations with a delay

                setTimeout(() => {

                    animatables.forEach((el, index) => {

                         if (el.classList.contains('fade-in')) {

                            el.style.transitionDelay = `${index * 150}ms`;

                            el.classList.add('fade-in');

                         }

                         if (el.classList.contains('slide-in-from-bottom')) {

                             el.classList.add('animate');

                         }

                         if (el.classList.contains('zoom-in')) {

                             el.classList.add('animate');

                         }

                    });

                }, 100); // Short delay to ensure class changes are registered

            }



            function goToNextSlide() {

                if (currentSlide < totalSlides) {

                    currentSlide++;

                    updateSlides();

                }

            }


            function goToPrevSlide() {

                if (currentSlide > 1) {

                    currentSlide--;

                    updateSlides();

                }

            }


            // --- Event Listeners ---

            nextBtn.addEventListener('click', goToNextSlide);

            prevBtn.addEventListener('click', goToPrevSlide);

            

            // Keyboard navigation

            document.addEventListener('keydown', (e) => {

                if (e.key === 'ArrowRight') {

                    goToNextSlide();

                } else if (e.key === 'ArrowLeft') {

                    goToPrevSlide();

                }

            });


            // --- Initial State ---

            updateSlides();

        });

    </script>
